void plutoPilot()
{

//Add your repeated code here

	if(Althold.getVelocityZ() > 0){

			ledOp(L_LEFT, ON);//green led
			ledOp(L_RIGHT, OFF);//red led
			//Print.monitor("Velocity Z:", Althold.getVelocityZ());

}
